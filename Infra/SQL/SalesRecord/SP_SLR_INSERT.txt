CREATE OR ALTER PROCEDURE SP_SLR_INSERT
(
	@P_DATE DATETIME,
	@P_AMOUNT DECIMAL(18, 2),
	@P_STATUS INTEGER,
	@P_SELER_ID INTEGER,
	@P_RETURN_VALUE INTEGER OUTPUT
)
AS
BEGIN
	INSERT INTO SALES_RECORD(SALE_DATE, SALE_AMOUNT, SALE_STATUS, SALE_SELLER_ID)
	 VALUES (@P_DATE, @P_AMOUNT, @P_STATUS, @P_SELER_ID);
	 
	 SELECT @P_RETURN_VALUE = MAX(SALE_ID) 
	   FROM SALES_RECORD
END;
