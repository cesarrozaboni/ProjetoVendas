CREATE OR ALTER PROCEDURE SP_SLR_UPDATE
(
	@P_ID	        INTEGER,
	@P_DATE         DATETIME,
	@P_AMOUNT       DECIMAL,
	@P_STATUS       INTEGER,
	@P_SELER_ID     INTEGER,
	@P_RETURN_VALUE INTEGER OUTPUT
)
AS
BEGIN
	UPDATE SALES_RECORD
	   SET SALE_DATE = @P_DATE,
		   SALE_AMOUNT = @P_AMOUNT,
		   SALE_STATUS = @P_STATUS,
		   SALE_SELLER_ID = @P_SELER_ID
	 WHERE SALE_ID = @P_ID;
	 
	 SET @P_RETURN_VALUE = @@ROWCOUNT;
END;
