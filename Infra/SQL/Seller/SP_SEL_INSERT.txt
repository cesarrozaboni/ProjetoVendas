CREATE OR ALTER PROCEDURE SP_SEL_INSERT
(
	@P_NAME           VARCHAR(100),
	@P_EMAIL          VARCHAR(40),
	@P_BIRTHDATE      DATETIME,
	@P_SALARY_BASE    DECIMAL(18,2),
	@P_DEPARTAMENT_ID INTEGER,
	@P_RETURN_VALUE   INTEGER OUTPUT
)
AS
BEGIN
	INSERT INTO SELLER(SELLER_NAME, SELLER_EMAIL, SELLER_BIRTHDATE, SELLER_SALARY_BASE, SELLER_ID_DEPARTAMENT)
	 VALUES (@P_NAME, @P_EMAIL, @P_BIRTHDATE, @P_SALARY_BASE, @P_DEPARTAMENT_ID);
	
	IF(@@ROWCOUNT >0)
		SELECT @P_RETURN_VALUE = MAX(SELLER_ID) FROM SELLER;
	ELSE 
		SET @P_RETURN_VALUE = @@ROWCOUNT;
END;
