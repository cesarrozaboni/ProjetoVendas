CREATE OR ALTER PROCEDURE SP_SEL_UPDATE
(
	@P_ID          INTEGER,
	@P_NAME        VARCHAR(100),
	@P_EMAIL       VARCHAR(40),
	@P_BIRTHDATE   DATETIME,
	@P_SALARY_BASE DECIMAL,
	@P_DEPARTAMENT_ID INTEGER,
	@P_RETURN_VALUE INTEGER OUTPUT
)
AS
BEGIN
	UPDATE SELLER
	   SET SELLER_NAME = @P_NAME,
	       SELLER_EMAIL = @P_EMAIL, 
		   SELLER_BIRTHDATE = @P_BIRTHDATE, 
		   SELLER_SALARY_BASE = @P_SALARY_BASE,
		   SELLER_ID_DEPARTAMENT = @P_DEPARTAMENT_ID
	 WHERE SELLER_ID = @P_ID;
	 
	 SET @P_RETURN_VALUE = @@ROWCOUNT;
END;
